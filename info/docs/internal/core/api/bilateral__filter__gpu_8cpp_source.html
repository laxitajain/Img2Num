41void bilateral_filter_gpu(
42 uint8_t* image, size_t width, size_t height, double sigma_spatial, double sigma_range,
43 uint8_t color_space
44) {
45 if (sigma_spatial <= 0.0 || sigma_range <= 0.0 || width <= 0 || height <= 0)
46 return;
47 if (color_space != COLOR_SPACE_OPTION_CIELAB && color_space != COLOR_SPACE_OPTION_RGB)
48 return;
49
50 std::vector<uint8_t> result(width * height * 4);
51 // copy image data to result incase filter fails
52 std::memcpy(result.data(), image, width * height * 4);
53 // CIELAB conversion will run as shader
54
55 std::cout << "begin wgpu portion" << std::endl;
56 // 1. Create Input Texture
57 wgpu::TextureDescriptor texDesc = {};
58 texDesc.size = {static_cast<uint32_t>(width), static_cast<uint32_t>(height), 1};
59 int bytesPerPixel {4};
60 texDesc.format = wgpu::TextureFormat::RGBA8Unorm;
61 texDesc.usage = wgpu::TextureUsage::TextureBinding | wgpu::TextureUsage::CopyDst;
62 wgpu::Texture inputTexture = GPU::getClassInstance().get_device().CreateTexture(&texDesc);
63
64 std::cout << "upload texture" << std::endl;
65 // Upload data to Input Texture
66 wgpu::TexelCopyTextureInfo dst = {};
67 dst.texture = inputTexture;
68 wgpu::TexelCopyBufferLayout layout = {};
69 layout.offset = 0;
70
71 layout.bytesPerRow = width * bytesPerPixel; // Tightly packed for upload
72 layout.rowsPerImage = height;
73 GPU::getClassInstance().get_queue().WriteTexture(
74 &dst, image, bytesPerPixel * width * height, &layout, &texDesc.size
75 );
76
77 std::cout << "create output texture" << std::endl;
78 // 2. Create Output Texture (Storage)
79 wgpu::TextureDescriptor outDesc = texDesc;
80 outDesc.usage = wgpu::TextureUsage::StorageBinding | wgpu::TextureUsage::CopySrc;
81 wgpu::Texture outputTexture = GPU::getClassInstance().get_device().CreateTexture(&outDesc);
82
83 // 2a. Intermediate textures for LAB if needed
84 wgpu::TextureDescriptor descLab = texDesc;
85 descLab.format = wgpu::TextureFormat::RGBA32Float; // <--- CRITICAL
86 descLab.usage = wgpu::TextureUsage::StorageBinding | wgpu::TextureUsage::TextureBinding;
87 // input lab
88 wgpu::Texture texLabRaw = GPU::getClassInstance().get_device().CreateTexture(&descLab);
89 // filtered lab
90 wgpu::Texture texLabFiltered = GPU::getClassInstance().get_device().CreateTexture(&descLab);
91
92 std::cout << "create buffer" << std::endl;
93 // 3. Create Uniform Buffer
94 float sr = static_cast<float>(sigma_range);
95 FilterParams params = {static_cast<float>(sigma_spatial), sr, 0.0f, 0.0f};
96 wgpu::BufferDescriptor bufDesc = {};
97 bufDesc.size = sizeof(FilterParams);
98 bufDesc.usage = wgpu::BufferUsage::Uniform | wgpu::BufferUsage::CopyDst;
99 wgpu::Buffer paramBuffer = GPU::getClassInstance().get_device().CreateBuffer(&bufDesc);
100 GPU::getClassInstance().get_queue().WriteBuffer(paramBuffer, 0, &params, sizeof(FilterParams));
101
102 wgpu::ComputePipeline pipeline;
103 wgpu::ComputePipeline pipelineRGB2LAB;
104 wgpu::ComputePipeline pipelineLAB2RGB;
105
106 switch (color_space) {
107 case COLOR_SPACE_OPTION_RGB: {
108 pipeline =
109 GPU::getClassInstance().createPipeline("bilateral_filter_rgb", "BilateralFilterShader");
110 break;
111 }
112 case COLOR_SPACE_OPTION_CIELAB: {
113 // also requires RGB-CIELAB conversion shaders
114 pipeline =
115 GPU::getClassInstance().createPipeline("bilateral_filter_lab", "BilateralFilterShader");
116 pipelineRGB2LAB = GPU::getClassInstance().createPipeline("rgb2cielab", "rgb2lab");
117 pipelineLAB2RGB = GPU::getClassInstance().createPipeline("cielab2rgb", "lab2rgb");
118 break;
119 }
120 }
121
122 // 6. Create Bind Group
123
124 // filter bind group
125 wgpu::BindGroupDescriptor bindGroupDesc = {};
126 bindGroupDesc.layout = pipeline.GetBindGroupLayout(0);
127 wgpu::BindGroupEntry entries[3];
128 // Entry 0: Input Texture View
129 entries[0].binding = 0;
130 entries[1].binding = 1;
131
132 switch (color_space) {
133 case COLOR_SPACE_OPTION_RGB: {
134 entries[0].textureView = inputTexture.CreateView();
135 // Entry 1: Output Texture View
136 entries[1].textureView = outputTexture.CreateView();
137 break;
138 }
139 case COLOR_SPACE_OPTION_CIELAB: {
140 entries[0].textureView = texLabRaw.CreateView();
141 // Entry 1: Output Texture View
142 entries[1].textureView = texLabFiltered.CreateView();
143 break;
144 }
145 }
146 // Entry 2: Uniform Buffer
147 entries[2].binding = 2;
148 entries[2].buffer = paramBuffer;
149 entries[2].size = sizeof(FilterParams);
150 bindGroupDesc.entryCount = 3;
151 bindGroupDesc.entries = entries;
152 wgpu::BindGroup bindGroup =
153 GPU::getClassInstance().get_device().CreateBindGroup(&bindGroupDesc);
154
155 wgpu::BindGroup bindGroupRGB2LAB;
156 wgpu::BindGroup bindGroupLAB2RGB;
157
158 if (color_space == COLOR_SPACE_OPTION_CIELAB) {
159 // rgb2lab bind group
160 wgpu::BindGroupEntry bg1Entries[2];
161 bg1Entries[0].binding = 0;
162 bg1Entries[0].textureView = inputTexture.CreateView();
163 bg1Entries[1].binding = 1;
164 bg1Entries[1].textureView = texLabRaw.CreateView();
165 wgpu::BindGroupDescriptor bg1Desc = {};
166 bg1Desc.layout = pipelineRGB2LAB.GetBindGroupLayout(0);
167 bg1Desc.entryCount = 2;
168 bg1Desc.entries = bg1Entries;
169 bindGroupRGB2LAB = GPU::getClassInstance().get_device().CreateBindGroup(&bg1Desc);
170 // lab2rgb bind group
171 wgpu::BindGroupEntry bg2Entries[2];
172 bg2Entries[0].binding = 0;
173 bg2Entries[0].textureView = texLabFiltered.CreateView();
174 bg2Entries[1].binding = 1;
175 bg2Entries[1].textureView = outputTexture.CreateView();
176 wgpu::BindGroupDescriptor bg2Desc = {};
177 bg2Desc.layout = pipelineLAB2RGB.GetBindGroupLayout(0);
178 bg2Desc.entryCount = 2;
179 bg2Desc.entries = bg2Entries;
180 bindGroupLAB2RGB = GPU::getClassInstance().get_device().CreateBindGroup(&bg2Desc);
181 }
182
183 // 7. Dispatch Compute Pass
184 wgpu::CommandEncoder encoder = GPU::getClassInstance().get_device().CreateCommandEncoder();
185
186 if (color_space == COLOR_SPACE_OPTION_CIELAB) {
187 wgpu::ComputePassEncoder pass1 = encoder.BeginComputePass();
188 pass1.SetPipeline(pipelineRGB2LAB);
189 pass1.SetBindGroup(0, bindGroupRGB2LAB);
190 // Workgroups of 16x16
191 pass1.DispatchWorkgroups((width + 15) / 16, (height + 15) / 16);
192 pass1.End();
193 }
194
195 wgpu::ComputePassEncoder pass = encoder.BeginComputePass();
196 pass.SetPipeline(pipeline);
197 pass.SetBindGroup(0, bindGroup);
198 // Workgroups of 16x16
199 pass.DispatchWorkgroups((width + 15) / 16, (height + 15) / 16);
200 pass.End();
201
202 if (color_space == COLOR_SPACE_OPTION_CIELAB) {
203 wgpu::ComputePassEncoder pass2 = encoder.BeginComputePass();
204 pass2.SetPipeline(pipelineLAB2RGB);
205 pass2.SetBindGroup(0, bindGroupLAB2RGB);
206 // Workgroups of 16x16
207 pass2.DispatchWorkgroups((width + 15) / 16, (height + 15) / 16);
208 pass2.End();
209 }
210
211 // 8. Prepare for Readback (Copy Texture -> Buffer)
212 // We cannot read textures directly on CPU. We must copy to a MapRead buffer.
213 uint32_t alignedBytesPerRow =
214 GPU::getAlignedBytesPerRow(width, static_cast<uint32_t>(bytesPerPixel));
215 uint32_t bufferSize = alignedBytesPerRow * height;
216
217 wgpu::BufferDescriptor readBufDesc = {};
218 readBufDesc.size = bufferSize;
219 readBufDesc.usage = wgpu::BufferUsage::MapRead | wgpu::BufferUsage::CopyDst;
220 wgpu::Buffer readBuffer = GPU::getClassInstance().get_device().CreateBuffer(&readBufDesc);
221
222 wgpu::TexelCopyTextureInfo srcTex = {};
223 srcTex.texture = outputTexture;
224
225 wgpu::TexelCopyBufferInfo dstBuf = {};
226 dstBuf.buffer = readBuffer;
227 dstBuf.layout.bytesPerRow = alignedBytesPerRow;
228
229 encoder.CopyTextureToBuffer(&srcTex, &dstBuf, &texDesc.size);
230
231 wgpu::CommandBuffer commands = encoder.Finish();
232 GPU::getClassInstance().get_queue().Submit(1, &commands);
233 std::cout << "queue submit" << std::endl;
234
235 // static volatile bool waiting = true;

237 // 9. Map Async (To read data back to C++)
238 // In a real app, you likely pass a callback function here.
239 struct ReadbackContext {
240 wgpu::Buffer buffer;
241 uint32_t size;
242 uint32_t alignedBytesPerRow;
243 int width;
244 int height;
245 };
246
247 uint8_t* result_ptr = result.data();
248 bool* waiting = new bool(true);
249
250 readBuffer.MapAsync(
251 wgpu::MapMode::Read, 0, bufferSize, wgpu::CallbackMode::AllowProcessEvents,
252 [](wgpu::MapAsyncStatus status, wgpu::StringView message, void* userdata) {
253 std::cout << "In callback" << std::endl;
254 bool* flag = static_cast<bool*>(userdata);
255 bool success = false;
256 if (status == wgpu::MapAsyncStatus::Success) {
257 success = true;
258 // std::cout << "Map success: " << message.data << std::endl;
259 } else {
260 // Handle error
261 success = false;
262 // std::cerr << "Map failed: " << message.data << std::endl;
263 }
264 *flag = false;
265 },
266 (void*)waiting
267 );
268
269 std::cout << "waiting " << *waiting << std::endl;
270
271 while (*waiting) {
272 GPU::getClassInstance().get_instance().ProcessEvents();
273#if defined(__EMSCRIPTEN__)
274 emscripten_sleep(10);
275#endif
276 }
277 std::cout << "done wgpu" << std::endl;
278 const uint8_t* mappedData = (const uint8_t*)readBuffer.GetConstMappedRange(0, bufferSize);
279 // copy to cpu buffer
280 for (size_t y = 0; y < height; ++y) {
281 const uint8_t* rowPtr = mappedData + (y * alignedBytesPerRow);
282 for (size_t x = 0; x < width; ++x) {
283 const uint8_t* pixelPtr = rowPtr + (x * bytesPerPixel);
284 size_t dstIndex = 4 * (y * width + x); // RGBA
285
286 std::memcpy(&result_ptr[dstIndex], pixelPtr, sizeof(uint8_t));
287 std::memcpy(&result_ptr[dstIndex + 1], pixelPtr + 1, sizeof(uint8_t));
288 std::memcpy(&result_ptr[dstIndex + 2], pixelPtr + 2, sizeof(uint8_t));
289 std::memcpy(&result_ptr[dstIndex + 3], pixelPtr + 3, sizeof(uint8_t));
290 }
291 }
292 readBuffer.Unmap();
293 std::memcpy(image, result.data(), result.size());
294 std::cout << "done memcpy" << std::endl;
295
296 // explicit clean up
297
298 if (inputTexture)
299 inputTexture.Destroy();
300 if (outputTexture)
301 outputTexture.Destroy();
302 if (texLabRaw)
303 texLabRaw.Destroy();
304 if (texLabFiltered)
305 texLabFiltered.Destroy();
306 readBuffer.Destroy();
307 delete waiting;
308#if defined(__EMSCRIPTEN__)
309 emscripten_sleep(50);
310#endif
311 result.clear();
312 result.shrink_to_fit();
313}
