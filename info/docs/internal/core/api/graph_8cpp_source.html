14static inline float
15colorDistance(const ImageLib::RGBPixel<uint8_t>& a, const ImageLib::RGBPixel<uint8_t>& b) {
16 ImageLib::RGBPixel<float> af {
17 static_cast<float>(a.red), static_cast<float>(a.green), static_cast<float>(a.blue)};
18 ImageLib::RGBPixel<float> bf {
19 static_cast<float>(b.red), static_cast<float>(b.green), static_cast<float>(b.blue)};
20 return std::sqrt(
21 (af.red - bf.red) * (af.red - bf.red) + (af.green - bf.green) * (af.green - bf.green) +
22 (af.blue - bf.blue) * (af.blue - bf.blue)
23 );
24}

66bool Graph::merge_nodes(const Node_ptr& node_to_keep, const Node_ptr& node_to_remove) {
67 auto end_node_ids {m_node_ids.end()};
68 auto node1_it {m_node_ids.find(node_to_keep->id())};
69 auto node2_it {m_node_ids.find(node_to_remove->id())};
70
71 if (node1_it == end_node_ids || node2_it == end_node_ids) {
72 return false;
73 }
74
75 const int32_t idx_k {node1_it->second};
76 const int32_t idx_r {node2_it->second};
77
78 // transfer edges from node_to_remove to node_to_keep
79 for (Node_ptr n : m_nodes->at(idx_r)->edges()) {
80 if (n->id() != node_to_keep->id()) {
81 // prevents self referencing
82 n->remove_edge(node_to_remove);
83 n->add_edge(node_to_keep);
84 node_to_keep->add_edge(n);

108void Graph::discover_edges(
109 const std::vector<int32_t>& region_labels, const int32_t width, const int32_t height
110) {
111 // Moore 8-connected neighbourhood
112 constexpr int8_t dirs[8][2] {{1, 0}, {-1, 0}, {0, 1}, {0, -1},
113 {1, 1}, {-1, -1}, {-1, 1}, {1, -1}};
114
115 int32_t rneigh[8];
116
117 for (int32_t y {0}; y < height; ++y) {
118 for (int32_t x {0}; x < width; ++x) {
119 const int32_t idx {y * width + x};
120 const int32_t rid {region_labels[idx]};
121
122 for (int32_t k {0}; k < 8; ++k) {
123 const int32_t nx {x + dirs[k][0]};
124 const int32_t ny {y + dirs[k][1]};
125
126 if (nx >= 0 && nx < width && ny >= 0 && ny < height) {
127 rneigh[k] = region_labels[ny * width + nx];
128 } else {
129 rneigh[k] = rid; // ignore out-of-bounds
130 }
131 }
132
133 for (int32_t r : rneigh) {
134 if (r != rid) {
135 add_edge(rid, r);
136 }
137 }
138 }
139 }

142void Graph::process_overlapping_edges() {
143 // 1. Build the Global Label Map ONCE (0 = background, else = node->id())
144 std::vector<int32_t> label_map(m_width * m_height, 0);
145
146 for (const Node_ptr& n : get_nodes()) {
147 if (n->area() == 0)
148 continue;
149
150 for (auto& [_, p] : n->get_pixels()) {
151 label_map[p.y * m_width + p.x] = n->id();
152 }
153 }
154
155 constexpr int8_t dirs[8][2] {{1, 0}, {-1, 0}, {0, 1}, {0, -1},
156 {1, 1}, {-1, -1}, {-1, 1}, {1, -1}};
157
158 // 2. Iterate directly over the pixels of each node
159 for (const Node_ptr& n : get_nodes()) {
160 if (n->area() == 0)
161 continue;
162 int32_t val = n->id();
163
164 for (auto& [_, p] : n->get_pixels()) {
165 int x = p.x;
166 int y = p.y;
167
168 // Check 8 neighbors in the global map
169 for (int k = 0; k < 8; ++k) {
170 int nx = x + dirs[k][0];
171 int ny = y + dirs[k][1];
172
173 // Fast boundary check (replaces std::clamp)
174 if (nx < 0 || nx >= m_width || ny < 0 || ny >= m_height)
175 continue;
176
177 int32_t n_val = label_map[ny * m_width + nx];
178
179 // Is it a neighbor? AND have we not processed this pairing yet?
180 if (n_val != 0 && n_val != val && val < n_val) {
181 bool is_too_thin = false;
182
183 // Check around the neighbor pixel for a 3rd region (pinching)
184 for (int mk = 0; mk < 8; ++mk) {
185 int mx = nx + dirs[mk][0];
186 int my = ny + dirs[mk][1];
187
188 if (mx < 0 || mx >= m_width || my < 0 || my >= m_height)
189 continue;
190
191 int32_t m_val = label_map[my * m_width + mx];
192
193 if (m_val != 0 && m_val != val && m_val != n_val) {
194 is_too_thin = true;
195 break; // CRITICAL: Stop checking immediately once proven thin!
196 }
197 }
198
199 if (is_too_thin) {
200 // Give our pixel to the neighbor
201 Node_ptr neighbor_node =
202 m_nodes->at(m_node_ids[n_val]); // get_node_by_id(n_val); // Assuming
203 // you have this lookup
204 if (neighbor_node) {
205 neighbor_node->add_edge_pixel(XY {x, y});
206 }
207 } else {
208 // Take the neighbor's pixel
209 n->add_edge_pixel(XY {nx, ny});
210 }
211 }
212 }
213 }
214 }

263void Graph::merge_small_area_nodes(const int32_t min_area) {
264 int32_t counter {0};
265 while (!all_areas_bigger_than(min_area)) {
266 for (const Node_ptr& n : get_nodes()) {
267 if (n->area() < min_area) {
268 std::vector<Node_ptr> neighbors;
269 neighbors.reserve(n->num_edges());
270 std::copy(n->edges().begin(), n->edges().end(), std::back_inserter(neighbors));
271
272 ImageLib::RGBPixel<uint8_t> col = n->color();
273
274 Node_ptr best_neighbor = nullptr;
275 float best_score = std::numeric_limits<float>::max();
276 for (const Node_ptr& ne : n->edges()) {
277 if (ne->area() > 0) {
278 float cdist = ImageLib::RGBPixel<uint8_t>::colorDistance(ne->color(), col);
279 float score = static_cast<float>(ne->area()) + 10.f * cdist;
280 if (score < best_score) {
281 best_score = score;
282 best_neighbor = ne;
283 }
284 }
285 }
286
287 // no valid neighbor found, skip this node
288 if (!best_neighbor) {
289 continue;
290 }
291
292 if (best_neighbor->area() >= n->area()) {
293 merge_nodes(best_neighbor, n);
294 } else {
295 merge_nodes(n, best_neighbor);
296 }
297 }
298 }
299
300 clear_unconnected_nodes();
301 ++counter;
302 }
